/** * Dazed Cegs - API de Consulta de Pedidos * ------------------------------------------------ * Le uma ou mais abas (CEGs) e expoe uma busca por @ da cliente. * * COMO PUBLICAR: * 1. Na planilha, va em Extensoes > Apps Script * 2. Apague o conteudo padrao e cole este arquivo inteiro * 3. Clique em Implantar > Nova implantacao * 4. Tipo: "App da Web" * 5. Executar como: "Eu" / Quem pode acessar: "Qualquer pessoa" * 6. Copie a URL gerada (termina em /exec) e cole no site * * COMO ADICIONAR UMA NOVA CEG (outra aba ou planilha): * Adicione um novo objeto na lista CEGS abaixo. Cada CEG aponta para * uma aba (sheetName) e diz se o filtro de POB deve aparecer para ela * (showPobFilter). Se a CEG estiver em outra planilha (arquivo), use * spreadsheetId para apontar para o ID daquele arquivo (entre /d/ e /edit * na URL do Google Sheets); deixe null para usar a planilha atual. */ const CEGS = [ { nome: 'POBS D:U', spreadsheetId: null, sheetName: 'POBS D:U', showPobFilter: true, imgFolder: 'pobs-du' }, { nome: 'POBS THE:SIN', spreadsheetId: '1CBOalqr1eNuJCPn1GLOgj-F_jNJ-X3LRSCKsgG1Q-ag', sheetName: 'POBS THE SIN', showPobFilter: true, imgFolder: 'pobs-the-sin' } ]; /** * Abas de frete e taxa. Cada uma representa um tipo de cobranca * (taxa de item, frete internacional, etc) com uma coluna "NOME + @", * quantidade, valor e uma coluna "VALIDACAO" (TRUE = pago). * Algumas abas tem duas cobrancas na mesma linha (frete E taxa * separados) -- por isso "campos" e uma lista. */ const CHARGE_SHEETS = [ { nome: 'Taxa 1ª leva', sheetName: 'TAXA LEVA PENDENTE', nameCol: 13, campos: [ { tipo: 'Taxa', qtyCol: 15, valorCol: 16, validacaoCol: 17 } ] }, { nome: 'Frete 1ª leva internacional', sheetName: '1 LEVA INTER - FRETE E TAXA', nameCol: 13, campos: [ { tipo: 'Frete', qtyCol: 15, valorCol: 16, validacaoCol: 17 } ] }, { nome: 'Frete e taxa 2ª leva internacional', sheetName: '2 LEVA INTER - FRETE E TAXA', nameCol: 12, campos: [ { tipo: 'Frete', qtyCol: 14, valorCol: 15, validacaoCol: 16 }, { tipo: 'Taxa', qtyCol: 14, valorCol: 17, validacaoCol: 18 } ] }, { nome: 'Leva Alemanha', sheetName: 'LEVA ITENS ALEMANHA', nameCol: 11, campos: [ { tipo: 'Frete + taxa', qtyCol: 13, valorCol: 14, validacaoCol: 15 } ] } ]; function doGet(e) { const action = (e.parameter.action || '').trim(); if (action === 'sobras') { return handleSobras_(); } const handleQuery = (e.parameter.at || '').trim(); const output = ContentService.createTextOutput(); output.setMimeType(ContentService.MimeType.JSON); if (!handleQuery) { output.setContent(JSON.stringify({ error: 'Informe o parametro @ (at).' })); return output; } const normalized = normalizeHandle(handleQuery); const allOrders = readAllOrdersFromAllCegs_(); const matches = allOrders.filter(o => o.handleNormalized === normalized); const allCharges = readAllCharges_(); const chargeMatches = allCharges.filter(c => c.handleNormalized === normalized); const itemsPendentes = matches.filter(m => m.pagamento === 'Pendente').length; output.setContent(JSON.stringify({ query: handleQuery, total: matches.length, cegs: CEGS.map(c => ({ nome: c.nome, showPobFilter: c.showPobFilter })), pedidos: matches.map(m => ({ ceg: m.ceg, imgFolder: m.imgFolder, pob: m.pob, pobSlug: slugifyPob_(m.pob), membro: m.membro, nome: m.nome, arroba: m.handle, valor: m.valor, prazo: m.prazo, pagamento: m.pagamento, status: m.status })), encargos: chargeMatches.map(c => ({ origem: c.origem, tipo: c.tipo, quantidade: c.quantidade, valor: c.valor, pagamento: c.pagamento })), resumoPendencias: { itensPendentes: itemsPendentes, encargosPendentes: chargeMatches.filter(c => c.pagamento === 'Pendente').length, temPendencia: itemsPendentes > 0 || chargeMatches.some(c => c.pagamento === 'Pendente') } })); return output; } function handleSobras_() { const output = ContentService.createTextOutput(); output.setMimeType(ContentService.MimeType.JSON); const sobras = readAllSobrasFromAllCegs_(); output.setContent(JSON.stringify({ total: sobras.length, itens: sobras.map(s => ({ ceg: s.ceg, imgFolder: s.imgFolder, pob: s.pob, pobSlug: slugifyPob_(s.pob), membro: s.membro, valor: s.valor, prazo: s.prazo, status: s.status })) })); return output; } function slugifyPob_(name) { return String(name || '') .toLowerCase() .normalize('NFD').replace(/[\u0300-\u036f]/g, '') .replace(/[^a-z0-9]+/g, '-') .replace(/^-+|-+$/g, ''); } function formatPrazo_(valor) { if (!valor) return ''; if (Object.prototype.toString.call(valor) === '[object Date]') { const dia = String(valor.getDate()).padStart(2, '0'); const mes = String(valor.getMonth() + 1).padStart(2, '0'); return dia + '/' + mes; } return String(valor).trim(); } function normalizeHandle(raw) { return String(raw) .replace(/^@/, '') .trim() .toLowerCase(); } function extractNameAndHandle_(rawJoiner) { const text = String(rawJoiner || '').trim(); // pega @ seguido (com ou sem espaco/tab) do identificador const match = text.match(/@\s*([A-Za-z0-9_.]+)/); let handle = ''; let nome = text; if (match) { handle = match[1]; nome = text.slice(0, match.index).replace(/[\t]+$/, '').trim(); } else { // fallback: sem @, assume ultima "palavra" como handle const parts = text.split(/\s+|\t+/).filter(Boolean); if (parts.length > 1) { handle = parts[parts.length - 1]; nome = parts.slice(0, -1).join(' '); } else { nome = text; } } return { nome: nome, handle: handle }; } function readAllOrdersFromAllCegs_() { let all = []; CEGS.forEach(ceg => { const ss = ceg.spreadsheetId ? SpreadsheetApp.openById(ceg.spreadsheetId) : SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName(ceg.sheetName); if (!sheet) return; const orders = readOrdersFromSheet_(sheet, ceg.nome, ceg.imgFolder); all = all.concat(orders); }); return all; } function readOrdersFromSheet_(sheet, cegNome, imgFolder) { const data = sheet.getDataRange().getValues(); const orders = []; let currentPob = ''; for (let i = 0; i < data.length; i++) { const row = data[i]; const colB = row[1]; // Membro ou data ou vazio const colC = row[2]; // Joiner + @ const colD = row[3]; // Telefone const colE = row[4]; // Valor const colF = row[5]; // Prazo const colG = row[6]; // Pagamento const colH = row[7]; // Status const colJ = row[9]; // Foto Itens (nome do POB) // linha de titulo do bloco: tem valor em J if (colJ && String(colJ).trim() !== '') { currentPob = String(colJ).trim(); } // pula cabecalhos e linhas vazias if (!colB || colB === 'Membro') continue; if (Object.prototype.toString.call(colB) === '[object Date]') continue; // linha so com data if (!colC) continue; const { nome, handle } = extractNameAndHandle_(colC); if (!handle) continue; orders.push({ ceg: cegNome, imgFolder: imgFolder || '', pob: currentPob, membro: String(colB).trim(), nome: nome, handle: '@' + handle, handleNormalized: normalizeHandle(handle), telefone: colD ? String(colD).trim() : '', valor: colE, prazo: formatPrazo_(colF), pagamento: colG === true ? 'Pago' : 'Pendente', status: colH ? String(colH).trim() : '' }); } return orders; } function readAllSobrasFromAllCegs_() { let all = []; CEGS.forEach(ceg => { const ss = ceg.spreadsheetId ? SpreadsheetApp.openById(ceg.spreadsheetId) : SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName(ceg.sheetName); if (!sheet) return; const sobras = readSobrasFromSheet_(sheet, ceg.nome, ceg.imgFolder); all = all.concat(sobras); }); return all; } function readSobrasFromSheet_(sheet, cegNome, imgFolder) { const data = sheet.getDataRange().getValues(); const sobras = []; let currentPob = ''; const membrosValidos = ['Jungwon', 'Heeseung', 'Jay', 'Jake', 'Sunghoon', 'Sunoo', 'Ni-ki']; for (let i = 0; i < data.length; i++) { const row = data[i]; const colB = row[1]; // Membro ou data ou vazio const colC = row[2]; // Joiner + @ const colE = row[4]; // Valor const colF = row[5]; // Prazo const colH = row[7]; // Status const colJ = row[9]; // Foto Itens (nome do POB) if (colJ && String(colJ).trim() !== '') { currentPob = String(colJ).trim(); } if (!colB) continue; if (Object.prototype.toString.call(colB) === '[object Date]') continue; const membroStr = String(colB).trim(); if (membrosValidos.indexOf(membroStr) === -1) continue; // ignora cabecalho, "set N", etc // sobra: tem membro valido mas a coluna Joiner+@ esta vazia if (colC && String(colC).trim() !== '') continue; sobras.push({ ceg: cegNome, imgFolder: imgFolder || '', pob: currentPob, membro: membroStr, valor: colE, prazo: formatPrazo_(colF), status: colH ? String(colH).trim() : '' }); } return sobras; } function readAllCharges_() { let all = []; CHARGE_SHEETS.forEach(cs => { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName(cs.sheetName); if (!sheet) return; const charges = readChargesFromSheet_(sheet, cs); all = all.concat(charges); }); return all; } function readChargesFromSheet_(sheet, cs) { const data = sheet.getDataRange().getValues(); const charges = []; for (let i = 0; i < data.length; i++) { const row = data[i]; const nameRaw = row[cs.nameCol - 1]; if (!nameRaw || nameRaw === 'NOME + @') continue; if (String(nameRaw).indexOf('@') === -1) continue; const { nome, handle } = extractNameAndHandle_(nameRaw); if (!handle) continue; cs.campos.forEach(campo => { const valor = row[campo.valorCol - 1]; if (valor === '' || valor === null || valor === undefined) return; const quantidade = row[campo.qtyCol - 1]; const validacao = row[campo.validacaoCol - 1]; charges.push({ origem: cs.nome, tipo: campo.tipo, nome: nome, handle: '@' + handle, handleNormalized: normalizeHandle(handle), quantidade: quantidade, valor: valor, pagamento: validacao === true ? 'Pago' : 'Pendente' }); }); } return charges; } /** * Funcao auxiliar para testar dentro do editor do Apps Script. * Selecione "testSearch" no menu de funcoes e clique em Executar. */ function testSearch() { const fake = { parameter: { at: 'mheesfv_' } }; const result = doGet(fake); Logger.log(result.getContent()); } /** * Funcao de diagnostico: lista as abas que o script enxerga * e o nome da planilha vinculada.
